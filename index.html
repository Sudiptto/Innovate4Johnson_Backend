Signup Canidate

Endpoint

POST /signupCanidate

Sample Input

{
    "firstName": "John",
    "lastName": "Doe",
    "username": "john_doe",
    "email": "john@example.com",
    "gradDate": "2027",
    "location": "New York",
    "password": "securepassword123"
}

HTTPS Request

curl -X POST http://localhost:5000/signupCanidate \
-H "Content-Type: application/json" \
-d '{
    "firstName": "John",
    "lastName": "Doe",
    "username": "john_doe",
    "email": "john@example.com",
    "gradDate": "2027",
    "location": "New York",
    "password": "securepassword123"
}'

Python

import requests

url = "http://localhost:5000/signupCanidate"
payload = {
    "firstName": "John",
    "lastName": "Doe",
    "username": "john_doe",
    "email": "john@example.com",
    "gradDate": "2027",
    "location": "New York",
    "password": "securepassword123"
}
response = requests.post(url, json=payload)
print(response.json())

TypeScript

import axios from 'axios';

const url = "http://localhost:5000/signupCanidate";
const payload = {
    firstName: "John",
    lastName: "Doe",
    username: "john_doe",
    email: "john@example.com",
    gradDate: "2027",
    location: "New York",
    password: "securepassword123"
};

axios.post(url, payload)
    .then(response => console.log(response.data))
    .catch(error => console.error(error));

Potential Output

{
    "message": "User created successfully"
}

Login Canidate

Endpoint

POST /loginCanidate

Sample Input

{
    "email": "john@example.com",
    "password": "securepassword123"
}

HTTPS Request

curl -X POST http://localhost:5000/loginCanidate \
-H "Content-Type: application/json" \
-d '{
    "email": "john@example.com",
    "password": "securepassword123"
}'

Python

import requests

url = "http://localhost:5000/loginCanidate"
payload = {
    "email": "john@example.com",
    "password": "securepassword123"
}
response = requests.post(url, json=payload)
print(response.json())
